Promises (goodbye callback hell)

Ejemplo descriptivo


            

//Promises

var request = require('request');

return new Promise((resolve, reject) => {
  request.get(url, (error, response, body) => {
    if (body) {
        resolve(JSON.parse(body));
      } else {
        resolve({});
      }
  });
});
              
              
              
              
//Promises paralelizadas

let urls = [
  '/api/commits',
  '/api/issues/opened',
  '/api/issues/assigned',
  '/api/issues/completed',
  '/api/issues/comments',
  '/api/pullrequests'
];

let promises = urls.map((url) => {
  return new Promise((resolve, reject) => {
    $.ajax({ url: url })
      .done((data) => {
        resolve(data);
      });
  });
});

Promise.all(promises)
  .then((results) => {
    // Do something with results of all our promises
 });

Descripcion

Los promises nos permiten convertir nuestro código horizontal (callback hell):

Características

  • El manejo de errores utilizando un montón de callbacks anidados puede ser caótico. Usando las promesas, tenemos un camino claro a los errores y podemos manejarlos apropiadamente.
  • El valor de una Promesa después de que ha sido resuelto / rechazado es inmutable - nunca cambiará.
  • También podemos paralelizar Promesas para manejar una matriz de operaciones asíncronas utilizando Promise.all ().
